(function () {
  'use strict';
  const COACH_VERSION = '1.0.0';

    if (!state || !state.activePbqId) return 'fb';

  const today = () => new Date().toISOString().slice(0, 10);
  function getCounter() {
    try {
      const raw = localStorage.getItem(COUNTER_KEY);
      if (!raw) return { date: today(), count: 0 };
      const parsed = JSON.parse(raw);
      if (parsed.date !== today()) return { date: today(), count: 0 };
      return parsed;

  const CACHE_KEY = 'tbV3CoachCache';
  const CACHE_TTL_MS = 24 * 60 * 60 * 1000;
  function cacheKey(input) {
    const s = JSON.stringify(input || {});
    let h = 5381;
    for (let i = 0; i < s.length; i++) h = ((h << 5) + h + s.charCodeAt(i)) | 0;
    return 'k' + (h >>> 0).toString(16);
  }
  function readCache() { try { return JSON.parse(localStorage.getItem(CACHE_KEY) || '{}'); } catch { return {}; } }
  function writeCache(o) { try { localStorage.setItem(CACHE_KEY, JSON.stringify(o)); } catch {} }
  function cacheGet(input) {
    const e = readCache()[cacheKey(input)];
    if (!e || Date.now() - e.t > CACHE_TTL_MS) return null;
    return e.v;
  }
  function cacheSet(input, value) {
    const c = readCache();
    c[cacheKey(input)] = { v: value, t: Date.now() };
    writeCache(c);
  }
```

  const PERSONA = `You are a Network+ (N10-009) tutor inside the Topology Builder. Keep responses under 500 tokens, calm and direct. Do not spoiler the PBQ answer when one is active. Stay on TB v3 topics.`;

  function buildPrompt(input) {
    const { mode, state = {}, stepId = null, hintsUsed = 0, certPack = 'netplus', aiPromptSeed = '', question = '' } = input;
    const stateJson = JSON.stringify({
      devices: (state.devices || []).map(d => ({ id: d.id, type: d.type, config: d.config })),
      cables: (state.cables || []).map(c => ({ from: c.from, to: c.to })),
    });
    const seed = aiPromptSeed.replace('{{state}}', stateJson);
    return [
      PERSONA,
      `Cert pack: ${certPack}.`,
      `Mode: ${mode}.`,
      stepId ? `Current step: ${stepId}. Hints used: ${hintsUsed}.` : '',
      seed,
      question ? `Student question: ${question}` : '',
    ].filter(Boolean).join('\n\n');
  }

  function defaultProvider() {
    throw new Error('AI provider not wired — supply { provider } or wait for integration (Task 17)');
  }

  function withTimeout(p, ms) {
    return Promise.race([p, new Promise((_, reject) => setTimeout(() => reject(new Error('timeout')), ms))]);
  }

  async function askAI(input, opts = {}) {
    const { provider = defaultProvider, timeoutMs = 10000 } = opts;
    const cached = cacheGet(input);
    if (cached) return cached;
    try {
      const result = await withTimeout(provider(buildPrompt(input)), timeoutMs);
      cacheSet(input, result);
      incrementCounter();
      return result;
    } catch (err) {
      if (err && err.status === 429) return 'BYOK quota reached for today.';
      return "I couldn't reach the tutor — try rephrasing, or check your connection.";
    }
  }
```
